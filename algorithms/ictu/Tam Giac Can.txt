package baitap;

import java.util.Scanner;

public class TamGiacCan {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int m = n * 2 - 1;

		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				if (i + j < n - 1 || j - i > n - 1) {
					System.out.print(" ");
				} else {
					System.out.print("*");
				}
			}
			System.out.println();

		}
	}
}
